import * as d3 from "d3";

create a rect

let svg = d3.create("svg");
let createarect = svg
  .attr("width", 100)
  .attr("height", 100)
  .call((svg) =>
    svg
      .selectAll("rect")
      .data([50])
      .enter()
      .append("rect")
      .attr("width", (d) => d)
      .attr("height", (d) => d)
      .attr("fill", "red")
  );
document.querySelector("#create-a-rect")?.append(createarect.node() as Node);

make it to center

.attr("x", (d) => d / 2)
.attr("y", (d) => d / 2)

or

.attr("transform", (d) => `translate(${d},${d / 2})`)
